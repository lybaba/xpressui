File Uploads

Champ fichier multiple avec drag-and-drop, validation et previews.

Code

import { mountFormUI } from "../dist/xpressui.mjs";

const container = document.getElementById("app");

mountFormUI(container, {
  version: 1,
  id: "upload_demo",
  uid: "upload_demo_uid",
  type: "contactform",
  name: "upload-demo",
  title: "Upload Demo",
  submit: {
    endpoint: "/api/uploads",
    mode: "form-data",
    formDataArrayMode: "repeat",
  },
  sections: {
    main: [
      { type: "text", name: "full_name", label: "Full Name", required: true },
      {
        type: "file",
        name: "attachments",
        label: "Attachments",
        multiple: true,
        minFiles: 1,
        maxFiles: 4,
        maxFileSizeMb: 5,
        maxTotalFileSizeMb: 12,
        accept: ".pdf,image/*",
        fileDropMode: "append",
        formDataFieldName: "attachments[]",
      },
    ],
  },
});
