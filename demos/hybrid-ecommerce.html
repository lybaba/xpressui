import { mountFormUI } from "../dist/xpressui.mjs";

const form = mountFormUI(app, {
  version: 1,
  id: "hybrid_ecommerce_demo",
  uid: "hybrid_ecommerce_demo_uid",
  type: "contactform",
  name: "hybrid-ecommerce-demo",
  title: "E-commerce Checkout",
  submit: {
    endpoint: "/api/orders",
  },
  sections: {
    main: [
      { type: "text", name: "customer_name", label: "Customer Name", required: true },
      { type: "email", name: "customer_email", label: "Email", required: true },
      {
        type: "select-multiple",
        name: "product_skus",
        label: "Products",
        required: true,
        choices: [
          { value: "sku-camera", label: "Nomad Camera Bag" },
          { value: "sku-headset", label: "Studio Headset" },
          { value: "sku-keyboard", label: "Mechanical Keyboard" },
          { value: "sku-lamp", label: "Desk Lamp" },
        ],
      },
      { type: "number", name: "quantity", label: "Quantity", required: true, min: 1 },
      { type: "text", name: "coupon", label: "Coupon" },
      { type: "textarea", name: "shipping_notes", label: "Shipping Notes" },
    ],
  },
});

form.setAttribute("mode", "hybrid");
form.setFieldOutputRenderer("product_skus", ({ value }) => {
  const container = document.createElement("ul");
  (Array.isArray(value) ? value : []).forEach((sku) => {
    const line = document.createElement("li");
    line.textContent = productsBySku[sku]?.name || sku;
    container.appendChild(line);
  });
  return container;
});
form.refreshOutputRendering();
