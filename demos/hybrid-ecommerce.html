E-commerce Checkout Multi-step

Exemple ecommerce multi-step avec product-list + image-gallery: étape 1 sélection produits, étape 2 checkout (résumé commande + total + paiement) + submit.

Code

import { createFormPreset, mountFormUI } from "../dist/xpressui.mjs";

const config = createFormPreset("ecommerce-checkout", {
  submit: { endpoint: "/api/mock-ecommerce-submit", method: "POST" }
});
const productsField = (config.sections.main || []).find((field) => field.name === "products");
if (productsField) {
  productsField.choices = products;
}
(config.sections.main || []).push({
  type: "image-gallery",
  name: "lookbook",
  label: "Lookbook Gallery",
  choices: lookbookImages
});

const mainFields = config.sections.main || [];
const stepOneFields = mainFields.filter((field) =>
  field.name === "products" || field.name === "lookbook"
);
const checkoutFields = [
  { type: "email", name: "billing_email", label: "Email", required: true },
  { type: "text", name: "payment_card_number", label: "Card Number", required: true },
  { type: "text", name: "payment_card_expiry", label: "Expiry (MM/YY)", required: true },
  { type: "text", name: "payment_card_cvc", label: "CVC", required: true }
];
config.type = "multistepform";
config.stepSections = [
  { type: "section", name: "catalog_step", label: "1. Cart & Details" },
  { type: "section", name: "checkout_step", label: "2. Checkout & Pay" },
];
config.sections = {
  ...(config.sections || {}),
  custom: [
    { type: "section", name: "catalog_step", label: "1. Cart & Details" },
    { type: "section", name: "checkout_step", label: "2. Checkout & Pay" },
  ],
  catalog_step: stepOneFields,
  checkout_step: checkoutFields,
};
config.stepLabels = {
  next: "Continue to checkout",
  previous: "Back to cart",
};

const form = mountFormUI(app, config);
