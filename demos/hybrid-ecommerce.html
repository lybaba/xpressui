Hybrid E-commerce Checkout

Exemple mode="hybrid" avec champ natif product-list (catalogue + cart global sticky à droite, max 20 produits sans pagination, clic produit pour galerie full).

Code

import { createFormPreset, mountFormUI } from "../dist/xpressui.mjs";

const config = createFormPreset("ecommerce-checkout");
const productsField = (config.sections.main || []).find((field) => field.name === "products");
if (productsField) {
  productsField.choices = products;
}

const form = mountFormUI(app, config);

form.setAttribute("mode", "hybrid");
form.setFieldOutputRenderer("products", ({ value }) => {
  const list = document.createElement("ul");
  (Array.isArray(value) ? value : []).forEach((product) => {
    const item = document.createElement("li");
    item.textContent = `${product.name} x${product.quantity}`;
    list.appendChild(item);
  });
  return list;
});
form.refreshOutputRendering();
