Approval Workflow

Démo d’un flux d’approbation avec panneau ops combiné branché sur les événements et l’état local.

Code

import { createFormOpsPanel, mountFormUI } from "../dist/xpressui.mjs";

const app = document.getElementById("app");
const debug = document.getElementById("debug");

const form = mountFormUI(app, {
  version: 1,
  id: "approval_demo",
  uid: "approval_demo_uid",
  type: "multistepform",
  name: "approval-demo",
  title: "Approval Demo",
  provider: {
    type: "approval-request",
    endpoint: "/api/approvals",
  },
  workflowStepTargets: {
    pending_approval: "review_step",
    approved: "review_step",
    completed: "review_step",
  },
  stepSections: [
    { type: "section", name: "request_step", label: "Request" },
    { type: "section", name: "review_step", label: "Review", stepSummary: true },
  ],
  sections: {
    custom: [
      { type: "section", name: "request_step", label: "Request" },
      { type: "section", name: "review_step", label: "Review", stepSummary: true },
    ],
    request_step: [
      { type: "email", name: "requester_email", label: "Requester Email", required: true },
      { type: "number", name: "amount", label: "Amount", required: true },
    ],
    review_step: [
      { type: "approval-state", name: "approval_state", label: "Approval Status" },
    ],
  },
});

const panel = createFormOpsPanel(form, form, {
  title: "Approval Ops",
  debug: { title: "Approval Debug" },
  admin: { title: "Approval Admin" },
});
debug.appendChild(panel.element);
